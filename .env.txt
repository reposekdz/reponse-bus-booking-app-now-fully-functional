# GoBus Environment Variables Configuration
# Copy the relevant sections into the correct .env files in your project.

# ===============================================
# == BACKEND CONFIGURATION (backend/.env file) ==
# ===============================================
# Create this file in the 'backend' directory for both development and production.

# --- Server Configuration ---
# Set to 'production' when deploying
NODE_ENV=development
PORT=5000

# --- Frontend URL ---
# IMPORTANT: For production, set this to the public URL of your frontend application (e.g., https://www.yourgobusapp.com)
# This is crucial for CORS and Socket.IO security.
FRONTEND_URL=http://localhost:3000

# --- Database Configuration ---
# Make sure your MySQL database is running and you have created a 'gobus' database.
DB_HOST=localhost
DB_USER=root
DB_PASSWORD=
DB_NAME=gobus

# --- JWT Authentication ---
# IMPORTANT: Use a long, random, and secure string for production.
# You can generate one using an online tool or `openssl rand -base64 32`
JWT_SECRET=your_super_secret_jwt_key_here_that_is_very_long

# --- Web Push Notifications (VAPID Keys) ---
# Generate these once using the command: npx web-push generate-vapid-keys
# The PRIVATE key stays here. The PUBLIC key goes into the frontend config.
# REQUIRED for production.
VAPID_PUBLIC_KEY=YOUR_GENERATED_VAPID_PUBLIC_KEY
VAPID_PRIVATE_KEY=YOUR_GENERATED_VAPID_PRIVATE_KEY
